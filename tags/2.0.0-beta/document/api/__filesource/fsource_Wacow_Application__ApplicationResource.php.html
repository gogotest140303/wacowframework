  98.     /**
  99.      * Get config object
  100.      *
  101.      * Here is an example to get a config object from '/path/to/config.ini' with ini type:
  102.      * <code>
  103.      * $setting = array(
  104.      *     'type'    => 'ini',
  105.      *     'path'    => '/path/to/config.ini',

  115.     public function getConfig($setting)
  116.     {
  117.         if ($setting instanceof Zend_Config) {
  118.             $setting = $setting->toArray();

  120.  
  121.         // params for build config object
  122.         $path    = null;

  134.  
  135.         // get config file type
  136.         if (isset($setting['type']) && in_array($setting['type'], $availableConfigType)) {
  137.             $type = ucfirst(strtolower($setting['type']));
  138.         }
  139.  
  140.         // get section name
  141.         if (isset($setting['section']) && is_string($setting['section'])) {
  142.             $section = $setting['section'];
  143.         }
  144.  
  145.         // if raw data given
  146.         if (isset($setting['data']) && is_array($setting['data'])) {

  148.         }
  149.  
  150.         // if config given

  154.

  159.  
  160.         // return config object

  163.         } else {
  164.             $className = 'Zend_Config_' . $type;
  165.             return new $className($path, $section, $config);
  166.         }
  167.     }

  190.     public function getDbAdapter($options)
  191.     {
  192.         // params for build adapter object
  193.         $type    = 'pdo_mysql';
  194.         $charset = 'utf8';
  195.  
  196.         if (is_array($options)) {

  198.             if (isset($options['type']) && is_string($options['type'])) {
  199.                 $type = strtolower($options['type']);
  200.             }
  201.  
  202.             // get adapter charset
  203.             if (isset($options['setting']['charset'])) {
  204.                 $charset = $options['setting']['charset'];
  205.             }
  206.  
  207.             // create adapter
  208.             $dbAdapter = Zend_Db::factory($type, $setting);
  209.         } elseif ($options instanceof Zend_Config) {
  210.             $dbAdapter = Zend_Db::factory($options);
  211.             $type = $options->adapter;
  212.             $charset = $options->params->charset;
  213.         }
  214.  
  215.         // correct mysql charset
  216.         // remove code below when Zend_Db resolved problem of mysql charset
  217.         if ((bool) preg_match('/mysql/i', $type) && isset($charset)) {
  218.             $dbAdapter->query('SET NAMES ' . $charset);
  219.         }
  220.  
  221.         return $dbAdapter;
  222.     }

  242.     public function getMailer($type, $options)
  243.     {
  244.         if (empty($type) || !is_string($type)) {
  245.             throw new Wacow_Application_Exception('Need the type of mailer.');
  246.         }
  247.  
  248.         // params for build adapter object
  249.         $server  = 'localhost';
  250.  
  251.         // convert to array
  252.         if ($options instanceof Zend_Config) {
  253.             $options = $options->toArray();
  254.         }
  255.  
  256.         // get smtp server
  257.         if (isset($options['server']) && is_string($options['server'])) {
  258.             $server = strtolower($options['server']);
  259.         }
  260.         unset($options['server']);
  261.  
  262.         // create mailer
  263.         $className = 'Wacow_Mail_' . $type;
  264.         $mailer = new $className($server, $options);
  265.  
  266.         return $mailer;
  267.     }
